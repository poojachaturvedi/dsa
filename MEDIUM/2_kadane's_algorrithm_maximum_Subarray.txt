 int maxSubArray(vector<int>& nums) {
        vector<int> subarray;
        int n=nums.size();
        int max_sum=0;
        for(int i=0;i<n;i++){
            for(int j=i;j<n;j++){
                int sum=0;
                for(int k=i;k<=j;j++){
                sum+=nums[k];
            }
            if(sum>max_sum){
                max_sum=sum;
                
            }
            }
        }
    return max_sum;
    }
